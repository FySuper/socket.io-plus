怎么使用?

添加两个必要文件!(must)

<script src="http://socketio.1473.cn/socket.io/socket.io.js"></script>
<script src="http://socketio.1473.cn/socket.js"></script>
<script >
            	var socket = new socket('qv');
            	//通过用户名,并获取socket对象
            	//特别注意的是socket.me 里面包含着这个用户的唯一id socket.me.id 
            	//切记要善用socket.me.id这个属性，可以在房间操作中更加灵活
</script>

加入房间

<script >
	//socket.join( 房间名, 回调函数 );					 
	//接受到消息的时候,回调函数会传回一个msg对象 --> { from: "" , data : "" } msg.from 来源 , msg.data 数据
	socket.join('game-11', function (msg) {
		console.log(msg);
		//{ from: xxx , data :xxx } 
	});
</script>

发送消息

<script >
 	socket.send(data, 'game-11');//向目标房间发送消息
</script>

获取房间信息

<script >
 	//向目标房间发送获取房间信息请求，并用附上回调函数接收
 	socket.list('game-11', function (msg) {
 		console.log(msg);
 		//打印出房间信息
 		//里面包含着这个房间有哪些成员
 	});
</script>

退出房间

<script >
 	//当你需要退出房间的时候，可以调用socket.exit(房间号)方法，也可以直接关闭浏览器。。他会检测并退出
 	socket.exit('game-11');
</script>
